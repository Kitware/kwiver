function (sprokit_add_python_run_test group instance)
  foreach (scheduler IN LISTS schedulers)
    sprokit_add_python_test(${group} ${instance}-${scheduler})

    set_tests_properties(test-python-${group}-${instance}-${scheduler}
      PROPERTIES
        TIMEOUT 5)
  endforeach ()
endfunction ()
